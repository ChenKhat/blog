排序算法​
  1. 选择排序法

    特点：

    • 原地排序
    • 不占用额外空间
function sort(arr) {
    for (let i = 0; i < arr.length; i++) {
        let minIndex = 0
        for (let j = i; j < arr.length; j++) {
            if(arr[j] < arr[minIndex])
                minIndex = j
        }
        swap(arr,i,minIndex)
    }
}

function swap(arr,index,minIndex) {
    const t = arr[index]
    arr[index] = arr[minIndex]
    arr[minIndex] = t
}
  1. 插入排序法

    特点

    • 原地排序
    • 不占用额外空间
function insertionSort(arr) {
    for (let i = 0; i < arr.length; i++) {
        const t = arr[i]
        for (let j = i; j - 1  > 1 ; j--) {
            if(arr[j] < arr[j-1]){
                arr[j] = arr[j-1]
            }
            arr[j-1] = t
        }
    }
}
